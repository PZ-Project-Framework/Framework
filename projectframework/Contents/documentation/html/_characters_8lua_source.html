22--! \brief Initialize a character.
23--! \return \string username
24function CHARACTER:Initialize()
25 if not self.playerObj then return end
26
27 local firstConnection = false
28 local characterModData = self.playerObj:getModData()["PFW_CHAR"] or nil
29
30 local inventory = ProjectFramework.Inventories:New(self.playerObj:getUsername())
31 self.inventoryID = inventory:Initialize()
32 self.inventory = inventory
33
34 if not characterModData then
35 firstConnection = true
36
37 self.playerObj:getModData()["PFW_CHAR"] = {
38 name = self.name or "Unknown",
39 description = self.description or "No description available.",
40 faction = self.faction or FACTION_CITIZEN,
41 age = self.age or 20,
42 heightFeet = self.heightFeet or 5,
43 heightInches = self.heightInches or 10,
44 eyeColor = self.eyeColor or "Brown",
45 hairColor = self.hairColor or "Brown",
46 physique = self.physique or "Average",
47 weight = self.weight or "125",
48 inventory = self.inventory.items or {},
49 upgrades = {}
50 }
51
52 if isClient() then
53 self.playerObj:transmitModData()
54 end
55 end
56
57 if firstConnection then
58 self:InitializeDefaultItems()
59 end
60
61 if isClient() then
62 timer:Simple(5, function()
63 self:ValidateCharacterData()
64 sendClientCommand("PFW_CHARS", "initialize", {self.playerObj:getUsername()})
65 end)
66 end
67
68 return ProjectFramework.Characters:Initialize(self.username, self)
69end
