21function class(BaseClass, ClassInitialiser)
22 local newClass = {} -- a new class newClass
23 if not ClassInitialiser and type(BaseClass) == 'function' then
24 ClassInitialiser = BaseClass
25 BaseClass = nil
26 elseif type(BaseClass) == 'table' then
27 -- our new class is a shallow copy of the base class!
28 for i,v in pairs(BaseClass) do
29 newClass[i] = v
30 end
31 newClass._base = BaseClass
32 end
33 -- the class will be the metatable for all its newInstanceects,
34 -- and they will look up their methods in it.
35 newClass.__index = newClass
36
37 -- expose a constructor which can be called by <classname>(<args>)
38 local classMetatable = {}
39 classMetatable.__call =
40 function(class_tbl, ...)
41 local newInstance = {}
42 setmetatable(newInstance,newClass)
43 --if init then
44 -- init(newInstance,...)
45 if class_tbl.init then
46 class_tbl.init(newInstance,...)
47 else
48 -- make sure that any stuff from the base class is initialized!
49 if BaseClass and BaseClass.init then
50 BaseClass.init(newInstance, ...)
51 end
52 end
53 return newInstance
54 end
55 newClass.init = ClassInitialiser
56 newClass.is_a =
57 function(this, klass)
58 local thisMetatable = getmetatable(this)
59 while thisMetatable do
60 if thisMetatable == klass then
61 return true
62 end
63 thisMetatable = thisMetatable._base
64 end
65 return false
66 end
67 setmetatable(newClass, classMetatable)
68 return newClass
69end

138--! \brief split a string
139--!
140--! \param Str
141--! \param Pattern
142--! \returns table of string fragments
143function string_split(Str, Pattern)
144 local splitStr = {}
145 local fpat = "(.-)" .. Pattern
146 local last_end = 1
147 local str, e, cap = string.find(Str,fpat, 1)
148 while str do
149 if str ~= 1 or cap ~= "" then
150 table.insert(splitStr,cap)
151 end
152 last_end = e+1
153 str, e, cap = string.find(Str,fpat, last_end)
154 end
155 if last_end <= #Str then
156 cap = string.sub(Str,last_end)
157 table.insert(splitStr, cap)
158 end
159 return splitStr
160end

314local function TString_removeCommentFromLine(Line)

324--! \brief get directive from magic
325local function getMagicDirective(Line)
326 local macro,tail
327 local macroStr = '[\\\@]'
328 local pos_macro = string.find(Line,macroStr)
329 if pos_macro then
330 --! ....\\ macro...stuff
331 --! ....\@ macro...stuff
332 local line = string.sub(Line,pos_macro+1)
333 local space = string.find(line,'%s+')
334 if space then
335 macro = string.sub(line,1,space-1)
336 tail = string_trim(string.sub(line,space+1))
337 else
338 macro = line
339 tail = ''
340 end
341 end
342 return macro,tail
343end

345--! \brief check comment for fn
346local function checkComment4fn(Fn_magic,MagicLines)
347 local fn_magic = Fn_magic
348-- TCore_IO_writeln('// checkComment4fn "' .. MagicLines .. '"')
349
350 local magicLines = string_split(MagicLines,'\n')
351
352 local macro,tail
353
354 for k,line in ipairs(magicLines) do
355 macro,tail = getMagicDirective(line)
356 if macro == 'fn' then
357 fn_magic = tail
358 -- TCore_IO_writeln('// found fn "' .. fn_magic .. '"')
359 else
360 --TCore_IO_writeln('// not found fn "' .. line .. '"')
361 end
362 end
363
364 return fn_magic
365end
366--! \brief run the filter
367function TLua2DoX_filter.readfile(this,AppStamp,Filename)
368 local err
369
370 local inStream = TStream_Read()
371 local outStream = TStream_Write()
372 this.outStream = outStream -- save to this obj
373
374 if (inStream:getContents(Filename)) then
375 -- output the file
376 local line
377 local fn_magic -- function name/def from magic comment
378
379 outStream:writelnTail('// #######################')
380 outStream:writelnTail('// app run:' .. AppStamp)
381 outStream:writelnTail('// #######################')
382 outStream:writelnTail()
383
384 while not (err or inStream:eof()) do
385 line = string_trim(inStream:getLine())
386-- TCore_Debug_show_var('inStream',inStream)
387-- TCore_Debug_show_var('line',line )
388 if string.sub(line,1,2)=='--' then -- its a comment
389 if string.sub(line,3,3)=='!' then -- it's a magic comment
390 local magic = string.sub(line,4)
391 outStream:writeln('
392 fn_magic = checkComment4fn(fn_magic,magic)
393 elseif string.sub(line,3,4)=='[[' then -- it's a long comment
394 line = string.sub(line,5) -- nibble head
395 local comment = ''
396 local closeSquare,hitend,thisComment
397 while (not err) and (not hitend) and (not inStream:eof()) do
398 closeSquare = string.find(line,']]')
399 if not closeSquare then -- need to look on another line
400 thisComment = line .. '\n'
401 line = inStream:getLine()
402 else
403 thisComment = string.sub(line,1,closeSquare-1)
404 hitend = true
405
406 -- unget the tail of the line
407 -- in most cases it's empty. This may make us less efficient but
408 -- easier to program
409 inStream:ungetLine(string_trim(string.sub(line,closeSquare+2)))
410 end
411 comment = comment .. thisComment
412 end
413 if string.sub(comment,1,1)=='!' then -- it's a long magic comment
414 outStream:write('/*' .. comment .. '*/ ')
415 fn_magic = checkComment4fn(fn_magic,comment)
416 else -- discard
417 outStream:write('/* zz:' .. comment .. '*/ ')
418 fn_magic = nil
419 end
420 else
421 outStream:writeln('// zz:"' .. line .. '"')
422 fn_magic = nil
423 end
424 elseif string.find(line,'^function') or string.find(line,'^local%s+function') then
425 -- it's a function
426 local pos_fn = string.find(line,'function')
427 -- function
428 -- ....v...
429 if pos_fn then
430 -- we've got a function
431 local fn_type
432 if string.find(line,'^local%s+') then
433 fn_type = 'static '
434 else
435 fn_type = ''
436 end
437 local fn = TString_removeCommentFromLine(string_trim(string.sub(line,pos_fn+8)))
438 if fn_magic then
439 fn = fn_magic
440 fn_magic = nil
441 end
442
443 if string.sub(fn,1,1)=='(' then
444 -- it's an anonymous function
445 outStream:writelnComment(line)
446 else
447 -- fn has a name, so is interesting
448
449 -- want to fix for iffy declarations
450 local open_paren = string.find(fn,'[%({]')
451 local fn0 = fn
452 if open_paren then
453 fn0 = string.sub(fn,1,open_paren-1)
454 -- we might have a missing close paren
455 if not string.find(fn,'%)') then
456 fn = fn .. ' ___MissingCloseParenHere___)'
457 end
458 end
459
460 local colon = string.find(fn0,'[%:]')
461 local dot = string.find(fn0,'[%.%:]')
462 local unusedStart, classStart = string.find(fn0,'.*[%.]')
463 unusedStart, classEnd = string.find(fn0,'.*[%.%:]')
464 if dot or (classStart and classEnd) then -- it's a method
465 local klass = classStart and classEnd and string.sub(fn,classStart,classEnd-1) or string.sub(fn,1,dot-1)
466 local method = string.sub(fn,dot+1)
467 --TCore_IO_writeln('function ' .. klass .. '::' .. method .. ftail .. '{}')
468 --TCore_IO_writeln(klass .. '::' .. method .. ftail .. '{}')
469 outStream:writeln('/*! \\memberof ' .. klass .. ' */ ' .. (colon and string.gsub(method, ":", "::") or method) .. '{}' )
470 else
471 -- add vanilla function
472
473 outStream:writeln(fn_type .. 'function ' .. fn .. '{}')
474 end
475 end
476 else
477 this:warning(inStream:getLineNo(),'something weird here')
478 end
479 fn_magic = nil -- mustn't indavertently use it again
480 elseif string.find(line,'=%s*class%(') then
481 -- it's a class declaration
482 local tailComment
483 line,tailComment = TString_removeCommentFromLine(line)
484 local equals = string.find(line,'=')
485 local klass = string_trim(string.sub(line,1,equals-1))
486 local tail = string_trim(string.sub(line,equals+1))
487 -- class(wibble wibble)
488 -- ....v.
489 local parent = string.sub(tail,7,-2)
490 if #parent>0 then
491 parent = ' :public ' .. parent
492 end
493 outStream:writeln('class ' .. klass .. parent .. '{};')
494 else
495 -- we don't know what this line means, so we can probably just comment it out
496 if #line>0 then
497 outStream:writeln('// zz: ' .. line)
498 else
499 outStream:writeln() -- keep this line blank
500 end
501 end
502 end
503
504 -- output the tail
505 outStream:write_tailLines()
506 else
507 outStream:writeln('!empty file')
508 end
509end
